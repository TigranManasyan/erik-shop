CREATE TABLE roles (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(30));
CREATE TABLE users ( id INT AUTO_INCREMENT PRIMARY KEY, first_name VARCHAR(40) NOT NULL, last_name VARCHAR(40) NOT NULL, phone VARCHAR(20) NOT NULL, avatar VARCHAR(100) DEFAULT 'avatar.png', email VARCHAR(100) UNIQUE NOT NULL, password VARCHAR(255) NOT NULL, role_id INT, FOREIGN KEY(role_id) REFERENCES roles(id), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP );

CREATE TABLE categories ( 
    id INT AUTO_INCREMENT PRIMARY KEY, 
    name VARCHAR(40) NOT NULL, 
    description TEXT, 
    user_id INT, 
    FOREIGN KEY(user_id) REFERENCES users(id), 
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP );
